        function formatDate(dateString) {
            if (!dateString) return 'N/A';
            try {
                const date = new Date(dateString);
                if (isNaN(date.getTime())) return 'Invalid Date';
                return date.toLocaleDateString('en-US', { 
                    month: 'short', 
                    day: 'numeric',
                    year: 'numeric'
                });
            } catch (error) {
                return 'N/A';
            }
        }

        function formatCurrency(amount) {
            return new Intl.NumberFormat('en-US', {
                style: 'currency',
                currency: 'USD'
            }).format(amount || 0);
        }

        async function loadOrders() {
            try {
                showLoading();
                console.log('Loading orders from:', ORDERS_API);
                
                // Try different API endpoints
                const endpoints = [
                    ORDERS_API,
                    `${API_BASE_URL}/api/order`,
                    `${API_BASE_URL}/api/orders/all`,
                    `${API_BASE_URL}/orders`
                ];
                
                let orders = [];
                let lastError = null;
                
                for (const endpoint of endpoints) {
                    try {
                        console.log('Trying endpoint:', endpoint);
                        const response = await fetch(endpoint, {
                            headers: getAuthHeaders()
                        });
                        
                        console.log('Response status:', response.status);
                        
                        if (response.ok) {
                            const data = await response.json();
                            console.log('API Response:', data);
                            
                            // Handle different response structures
                            if (Array.isArray(data)) {
                                orders = data;
                                console.log('Got orders array directly:', orders.length);
                                break;
                            } else if (data.data && Array.isArray(data.data)) {
                                orders = data.data;
                                console.log('Got orders from data property:', orders.length);
                                break;
                            } else if (data.orders && Array.isArray(data.orders)) {
                                orders = data.orders;
                                console.log('Got orders from orders property:', orders.length);
                                break;
                            } else if (typeof data === 'object') {
                                // If it's a single order object
                                orders = [data];
                                console.log('Got single order object');
                                break;
                            }
                        } else if (response.status === 401) {
                            localStorage.removeItem('adminAuthToken');
                            localStorage.removeItem('token');
                            showAlert('Session expired. Please login again.', 'warning');
                            setTimeout(() => {
                                window.location.href = 'login.html';
                            }, 2000);
                            return;
                        }
                    } catch (error) {
                        lastError = error;
                        console.log(`Endpoint ${endpoint} failed:`, error.message);
                    }
                }
                
                if (orders.length === 0) {
                    // Try one more time without auth headers (for testing)
                    try {
                        console.log('Trying without auth headers...');
                        const response = await fetch(ORDERS_API);
                        if (response.ok) {
                            const data = await response.json();
                            if (Array.isArray(data)) {
                                orders = data;
                            } else if (data.data && Array.isArray(data.data)) {
                                orders = data.data;
                            }
                        }
                    } catch (error) {
                        console.log('Failed without auth too:', error);
                    }
                }
                
                if (orders.length === 0) {
                    // Use demo data if no orders found
                    console.log('No orders found from API, using demo data');
                    orders = getDemoOrders();
                    showAlert('Using demo data. Connect to your backend to see real orders.', 'info');
                } else {
                    showAlert(`Successfully loaded ${orders.length} orders`, 'success');
                }
                
                // Store and display orders
                currentOrders = orders;
                filteredOrders = [...currentOrders];
                
                displayOrders();
                updateStats();
                
            } catch (error) {
                console.error('Error loading orders:', error);
                showAlert(`Error loading orders: ${error.message}`, 'danger');
                
                // Fallback to demo data
                currentOrders = getDemoOrders();
                filteredOrders = [...currentOrders];
                displayOrders();
                updateStats();
                showAlert('Loaded demo orders', 'info');
            } finally {
                hideLoading();
            }
        }

        function initializeEventListeners() {
            // Filter buttons

        function applyFilters() {
            let filtered = [...currentOrders];
            const searchTerm = elements.globalSearch.value.toLowerCase();
            
            // Search filter
            if (searchTerm) {
                filtered = filtered.filter(order => {
                    const orderId = (order._id || order.id || order.orderId || '').toString().toLowerCase();
                    const customerName = (order.customer?.name || order.user?.name || 'Guest').toLowerCase();
                    const customerEmail = (order.customer?.email || order.user?.email || '').toLowerCase();
                    
                    return orderId.includes(searchTerm) ||
                           customerName.includes(searchTerm) ||
                           customerEmail.includes(searchTerm);
                });
            }
            
            // Status filter
            const statusValue = elements.statusFilter.value;
            if (statusValue) {
                filtered = filtered.filter(order => 
                    order.status && order.status.toLowerCase() === statusValue.toLowerCase()
                );
            }
            
            // Date filter
            const dateValue = elements.dateFilter.value;
            if (dateValue) {
                const now = new Date();
                let startDate;
                
                switch(dateValue) {
                    case 'today':
                        startDate = new Date(now.getFullYear(), now.getMonth(), now.getDate());
                        break;
                    case 'week':
                        startDate = new Date(now.getTime() - 7 * 24 * 60 * 60 * 1000);
                        break;
                    case 'month':
                        startDate = new Date(now.getFullYear(), now.getMonth(), 1);
                        break;
                }
                
                if (startDate) {
                    filtered = filtered.filter(order => {
                        const orderDate = new Date(order.createdAt || order.orderDate);
                        return orderDate >= startDate;
                    });
                }
            }
            
            filteredOrders = filtered;
            displayOrders();
        }

        async function viewOrderDetails(orderId) {

                showLoading();

                    (o._id || o.id || o.orderId) === orderId
                );
                
                if (!order) {
                    showAlert('Order not found', 'warning');
                    return;
                }
                
                // Build order details HTML
                const customerName = order.customer?.name || order.user?.name || 'Guest Customer';
                const customerEmail = order.customer?.email || order.user?.email || '';
                const shippingAddress = order.shippingAddress || {};
                const items = order.items || order.products || [];
                const totalAmount = order.totalAmount || order.total || 0;
                const status = order.status || 'pending';
                const orderDate = order.createdAt || order.orderDate;
                
                let itemsHtml = '';
                if (Array.isArray(items) && items.length > 0) {
                    itemsHtml = items.map(item => `
                        <tr>
                            <td>${item.name || 'Unknown Product'}</td>
                            <td class="text-center">${item.quantity || 1}</td>
                            <td class="text-end">${formatCurrency(item.price || 0)}</td>
                            <td class="text-end">${formatCurrency((item.price || 0) * (item.quantity || 1))}</td>
                        </tr>
                    `).join('');
                } else {
                    itemsHtml = '<tr><td colspan="4" class="text-center">No items found</td></tr>';
                }
                
                const detailsHtml = `

                        <div class="col-md-6">
                            <h6>Order Information</h6>
                            <table class="table table-sm">
                                <tr><td><strong>Order ID:</strong></td><td>${orderId}</td></tr>
                                <tr><td><strong>Order Date:</strong></td><td>${formatDate(orderDate)}</td></tr>
                                <tr><td><strong>Status:</strong></td><td><span class="status-badge ${status}">${status}</span></td></tr>
                                <tr><td><strong>Total Amount:</strong></td><td><strong>${formatCurrency(totalAmount)}</strong></td></tr>
                            </table>
                        </div>
                        <div class="col-md-6">
                            <h6>Customer Information</h6>
                            <table class="table table-sm">
                                <tr><td><strong>Name:</strong></td><td>${customerName}</td></tr>
                                <tr><td><strong>Email:</strong></td><td>${customerEmail}</td></tr>
                                ${shippingAddress.address ? `
                                    <tr><td><strong>Address:</strong></td><td>${shippingAddress.address}</td></tr>
                                    <tr><td><strong>City:</strong></td><td>${shippingAddress.city || ''}</td></tr>
                                    <tr><td><strong>Country:</strong></td><td>${shippingAddress.country || ''}</td></tr>
                                ` : ''}

                            <thead>
                                <tr>
                                    <th>Product</th>
